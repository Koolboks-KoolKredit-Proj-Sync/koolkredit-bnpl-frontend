import React, { useState, useEffect, useRef } from "react";
import { Shield, User, AlertCircle, Camera, CheckCircle, X } from "lucide-react";
import Swal from "sweetalert2";
import { useParams, useNavigate } from "react-router-dom";

export default function GuarantorFormPage() {
  const { token } = useParams();
  const navigate = useNavigate();

  const [isLoading, setIsLoading] = useState(true);
  const [isSubmitting, setIsSubmitting] = useState(false);
  const [customerData, setCustomerData] = useState(null);
  const [error, setError] = useState(null);

  const [formData, setFormData] = useState({
    bvn: "",
    phoneNumber: "",
    nin: "",
    selfieImage: null,
  });

  /* -------------------- SELFIE STATES -------------------- */
  const [showCamera, setShowCamera] = useState(false);
  const [stream, setStream] = useState(null);
  const [capturedImage, setCapturedImage] = useState(null);
  const [liveness, setLiveness] = useState({
    status: "idle",
    instruction: "",
  });

  const videoRef = useRef(null);
  const canvasRef = useRef(null);

  /* -------------------- FETCH CUSTOMER -------------------- */
  useEffect(() => {
    const fetchCustomerData = async () => {
      try {
        const res = await fetch(
          `http://localhost:8080/api/guarantor/form/${token}`
        );
        const data = await res.json();

        if (!data.success) throw new Error(data.message);
        setCustomerData(data);
      } catch (err) {
        setError(err.message || "Invalid link");
      } finally {
        setIsLoading(false);
      }
    };

    fetchCustomerData();
  }, [token]);

  /* -------------------- FORM HANDLERS -------------------- */
  const handleChange = (e) => {
    const { name, value } = e.target;
    if (!/^\d*$/.test(value)) return;
    if ((name === "bvn" || name === "nin") && value.length > 11) return;
    setFormData((p) => ({ ...p, [name]: value }));
  };

  /* -------------------- CAMERA -------------------- */
  const startCamera = async () => {
    try {
      const mediaStream = await navigator.mediaDevices.getUserMedia({
        video: { facingMode: "user" },
      });

      setStream(mediaStream);
      videoRef.current.srcObject = mediaStream;
      setShowCamera(true);
      startLiveness();
    } catch {
      Swal.fire("Camera Error", "Camera access denied", "error");
    }
  };

  const stopCamera = () => {
    stream?.getTracks().forEach((t) => t.stop());
    setStream(null);
    setShowCamera(false);
  };

  /* -------------------- LIVENESS (UI BASED) -------------------- */
  const startLiveness = () => {
    const steps = [
      "Look straight at the camera",
      "Blink twice",
      "Turn head left",
      "Turn head right",
      "Smile 😊",
    ];

    let step = 0;
    setLiveness({ status: "checking", instruction: steps[0] });

    const interval = setInterval(() => {
      step++;
      if (step === steps.length) {
        clearInterval(interval);
        setLiveness({ status: "passed", instruction: "Verified!" });
        setTimeout(captureSelfie, 800);
      } else {
        setLiveness({ status: "checking", instruction: steps[step] });
      }
    }, 2000);
  };

  /* -------------------- CAPTURE -------------------- */
  const captureSelfie = () => {
    const video = videoRef.current;
    const canvas = canvasRef.current;

    canvas.width = video.videoWidth;
    canvas.height = video.videoHeight;
    canvas.getContext("2d").drawImage(video, 0, 0);

    canvas.toBlob((blob) => {
      setFormData((p) => ({ ...p, selfieImage: blob }));
      setCapturedImage(URL.createObjectURL(blob));
      stopCamera();
    }, "image/jpeg");
  };

  const retakeSelfie = () => {
    setCapturedImage(null);
    setFormData((p) => ({ ...p, selfieImage: null }));
    startCamera();
  };

  /* -------------------- VALIDATION -------------------- */
  const validateForm = () => {
    if (formData.bvn.length !== 11)
      return Swal.fire("Error", "BVN must be 11 digits", "error");
    if (formData.nin.length !== 11)
      return Swal.fire("Error", "NIN must be 11 digits", "error");
    if (!formData.selfieImage)
      return Swal.fire("Error", "Selfie verification required", "error");
    return true;
  };

  /* -------------------- SUBMIT -------------------- */
  const handleSubmit = async (e) => {
    e.preventDefault();
    if (!validateForm()) return;

    setIsSubmitting(true);

    try {
      const payload = new FormData();
      payload.append("guarantorBvn", formData.bvn);
      payload.append("guarantorPhoneNumber", formData.phoneNumber);
      payload.append("guarantorNin", formData.nin);
      payload.append("selfieImage", formData.selfieImage);

      const res = await fetch(
        `http://localhost:8080/api/guarantor/submit/${token}`,
        {
          method: "POST",
          body: payload,
        }
      );

      const data = await res.json();
      if (!data.success) throw new Error(data.message);

      Swal.fire("Success", "Submitted successfully", "success").then(() =>
        navigate(`/guarantor/otp/${token}`)
      );
    } catch (err) {
      Swal.fire("Error", err.message, "error");
    } finally {
      setIsSubmitting(false);
    }
  };

  /* -------------------- UI -------------------- */
  if (isLoading) return <div className="text-center">Loading...</div>;
  if (error) return <div className="text-center text-red-500">{error}</div>;

  return (
    <div className="min-h-screen p-6" style={{ background: "#f7623b" }}>
      <div className="max-w-3xl mx-auto bg-black p-8 rounded-xl">
        <h1 className="text-2xl text-[#f7623b] font-bold mb-6">
          Guarantor Verification
        </h1>

        {customerData && (
            <div className="bg-gray-900 rounded-lg p-6 border border-gray-800 mb-6">
              <h2 className="text-lg font-semibold text-white mb-4">
                Customer Information
              </h2>
          
              <div className="grid grid-cols-1 sm:grid-cols-2 gap-4 text-sm">
                <div>
                  <p className="text-gray-400">Customer Name</p>
                  <p className="text-white font-medium">
                    {customerData.customerFirstName} {customerData.customerLastName}
                  </p>
                </div>
          
                <div>
                  <p className="text-gray-400">Plan</p>
                  <p className="text-white font-medium">
                    {customerData.customerPlan}
                  </p>
                </div>
          
                <div>
                  <p className="text-gray-400">BVN</p>
                  <p className="text-white font-medium">
                    {customerData.customerBvn}
                  </p>
                </div>
          
                <div>
                  <p className="text-gray-400">Duration</p>
                  <p className="text-white font-medium">
                    {customerData.customerInstallmentDuration}
                  </p>
                </div>
              </div>
            </div>
          )}
          

        <form onSubmit={handleSubmit} className="space-y-6">
          {/* INPUTS */}
          {["bvn", "phoneNumber", "nin"].map((f) => (
            <input
              key={f}
              name={f}
              value={formData[f]}
              onChange={handleChange}
              placeholder={f.toUpperCase()}
              className="w-full p-3 bg-gray-900 text-white rounded"
              required
            />
          ))}

          {/* SELFIE */}
          {!capturedImage && !showCamera && (
            <button
              type="button"
              onClick={startCamera}
              className="w-full bg-[#f7623b] p-3 rounded text-white"
            >
              <Camera className="inline mr-2" /> Start Selfie Verification
            </button>
          )}

          {showCamera && (
            <div className="relative">
              <video ref={videoRef} autoPlay className="rounded w-full" />
              <div className="absolute top-2 left-2 bg-black/60 text-white px-3 py-1 rounded">
                {liveness.instruction}
              </div>
              <button
                onClick={stopCamera}
                className="absolute top-2 right-2 bg-red-500 p-2 rounded"
              >
                <X />
              </button>
            </div>
          )}

          {capturedImage && (
            <div>
              <img src={capturedImage} className="rounded border-green-500" />
              <button
                type="button"
                onClick={retakeSelfie}
                className="mt-2 bg-gray-700 p-2 rounded w-full text-white"
              >
                Retake Selfie
              </button>
            </div>
          )}

          <canvas ref={canvasRef} className="hidden" />

          <button
            type="submit"
            disabled={isSubmitting}
            className="w-full bg-[#f7623b] p-4 rounded text-white font-bold"
          >
            {isSubmitting ? "Submitting..." : "Submit"}
          </button>
        </form>
      </div>
    </div>
  );
}
